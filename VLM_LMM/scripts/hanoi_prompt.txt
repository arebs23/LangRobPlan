I want you to solve a planning problem. Your tasks are:

1. Generate the goal state in PDDL format based on user input in form of natural language.
2. Generate the initial state in PDDL format based on the provided image and the objects detected in the image.

An example is:

(define (problem hanoi1)
    (:domain hanoi)
    (:objects
        peg1
        peg2
        peg3
        green_disk
        blue_disk
        pink_disk
    )
        (:init
            (clear green_disk)
            (clear peg2)
            (clear peg3)
            (on green_disk blue_disk)
            (on blue_disk pink_disk)
            (on pink_disk peg1)
            (smaller green_disk peg1)
            (smaller blue_disk peg1)
            (smaller pink_disk peg1)
            (smaller green_disk peg2)
            (smaller blue_disk peg2)
            (smaller pink_disk peg2)
            (smaller green_disk peg3)
            (smaller blue_disk peg3)
            (smaller pink_disk peg3)
            (smaller green_disk blue_disk)
            (smaller green_disk pink_disk)
            (smaller blue_disk pink_disk)
            
        )
    (:goal (and (on pink_disk peg3) (on blue_disk pink_disk) (on green_disk blue_disk)))
)
